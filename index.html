function doGet(e) { var action = e.parameter.action; var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; if (action === "read") { var rows = sheet.getDataRange().getValues(); var data = []; // 첫 줄(헤더) 제외하고 데이터 추출 for (var i = 1; i < rows.length; i++) { if(rows[i][0]) { data.push({ vendor: rows[i][0].toString(), cartype: rows[i][1].toString(), partname: rows[i][2].toString(), code: rows[i][3].toString() }); } } // CORS 에러 방지를 위해 TextOutput에 헤더를 명시적으로 설정하여 리턴 return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } } function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheets()[0]; var postData = JSON.parse(e.postData.contents); sheet.clear(); sheet.appendRow(["업체명", "차종", "자재명", "코드"]); // 헤더 삽입 if (Array.isArray(postData)) { for (var i = 0; i < postData.length; i++) { sheet.appendRow([ postData[i].vendor, postData[i].cartype, postData[i].partname, postData[i].code ]); } } // 성공 응답도 동일하게 리턴 return ContentService.createTextOutput("Success") .setMimeType(ContentService.MimeType.TEXT); }
